--查看版本
npm -v
node -v

--查看global模块
npm list -g/-global   (-g : global)

--初始化
# install dependencies
cnpm install  （淘宝镜像）
npm install


--启动项目
# serve with hot reload at localhost:8080
  cnpm run dev (淘宝镜像)
  npm run dev

  # build for production with minification
  npm run build

  # build for production and view the bundle analyzer report
  npm run build --report

  # run unit tests
  npm run unit

  # run e2e tests
  npm run e2e

  # run all tests
  npm test


--检查配置的镜像是否可用
npm config get registry
npm info vue
